1. Socket 基础概念 Fundamentals

Socket（套接字） = 进程间通信的端点。两个 socket 配对形成一个连接，应用程序通过 socket 把数据交给传输层。

Socket 由三元组唯一标识

  • IP 地址（互联网中机器的标识）
  • 端口号 Port（1–65535，标识机器上具体进程）
  • 协议（TCP 或 UDP）

Two Socket Types 两种类型

类型使用协议特点API
Stream Socket
流套接字		TCP可靠、有序的字节流SOCK_STREAM
Datagram Socket
数据报套接字		UDP不可靠、独立数据包，每包≤65,500 字节SOCK_DGRAM

知名端口 Well-Known Ports (1–1023)

端口服务端口服务
20/21FTP53DNS
22SSH80HTTP
23Telnet443HTTPS
25SMTP110POP3

应用自用端口一般选 1024–65535。

Client-Server 模型

  • Server：被动等待，监听特定端口；不预先知道有哪些客户端
  • Client：主动发起，必须知道服务器的 IP + Port

3. TCP Server 完整代码 ⭐⭐ 必背

编译时记得链接 Winsock 库：gcc server.c -o server -lws2_32

// tcp_server.c — Windows / Winsock
#include <stdio.h>
#include <string.h>
#include <winsock2.h>
#pragma comment(lib, "ws2_32.lib")    // 链接 Winsock 库

#define PORT 7000
#define BUFSIZE 1024

int main(void) {
    WSADATA wsaData;
    SOCKET serverSock, clientSock;
    struct sockaddr_in serverAddr, clientAddr;
    int addrLen = sizeof(clientAddr);
    char buffer[BUFSIZE];
    int recvBytes;

    // 1) 初始化 Winsock（请求 2.2 版本）
    if (WSAStartup(MAKEWORD(2, 2), &wsaData) != 0) {
        fprintf(stderr, "WSAStartup failed\n");
        return 1;
    }

    // 2) 创建 socket：IPv4 + TCP
    serverSock = socket(AF_INET, SOCK_STREAM, 0);
    if (serverSock == INVALID_SOCKET) {
        perror("socket"); WSACleanup(); return 1;
    }

    // 3) 准备地址结构
    serverAddr.sin_family = AF_INET;
    serverAddr.sin_addr.s_addr = htonl(INADDR_ANY);  // 任意网卡
    serverAddr.sin_port = htons(PORT);                // 主机字节序 → 网络字节序

    // 4) 绑定 socket 到端口
    if (bind(serverSock, (struct sockaddr*)&serverAddr, sizeof(serverAddr)) == SOCKET_ERROR) {
        perror("bind"); closesocket(serverSock); WSACleanup(); return 1;
    }

    // 5) 监听，最多 5 个挂起连接
    if (listen(serverSock, 5) == SOCKET_ERROR) {
        perror("listen"); closesocket(serverSock); WSACleanup(); return 1;
    }
    printf("Server listening on port %d...\n", PORT);

    // 6) 接受连接（阻塞）
    clientSock = accept(serverSock, (struct sockaddr*)&clientAddr, &addrLen);
    if (clientSock == INVALID_SOCKET) {
        perror("accept"); closesocket(serverSock); WSACleanup(); return 1;
    }
    printf("Client connected.\n");

    // 7) 收发数据
    recvBytes = recv(clientSock, buffer, BUFSIZE - 1, 0);
    if (recvBytes > 0) {
        buffer[recvBytes] = '\0';
        printf("Received: %s\n", buffer);
        send(clientSock, "Hello from server!", 18, 0);
    }

    // 8) 关闭
    closesocket(clientSock);
    closesocket(serverSock);
    WSACleanup();
    return 0;
}
💡 每个函数到底干啥？中文逐行
  • WSAStartup：Windows 特有，加载 Winsock DLL，告诉系统你要用网络
  • socket(AF_INET, SOCK_STREAM, 0)：申请一个 IPv4 + TCP 的套接字
  • sockaddr_in：地址结构体，含族 (AF_INET)、地址、端口
  • htons / htonl：把主机字节序转成网络字节序（大端）。必须用，否则端口会错
  • INADDR_ANY：绑定到本机所有 IP，不限定某张网卡
  • bind：把 socket 关联到那个地址，从此该端口的数据都送给这个 socket
  • listen(s, backlog)：把 socket 切到被动监听态，backlog 是内核队列长度
  • accept：阻塞等连接，返回一个新 socket 专门和该客户端通信
  • recv/send：在已连接 socket 上收发
  • closesocket / WSACleanup：释放资源

4. TCP Client 完整代码 ⭐⭐

// tcp_client.c — Windows / Winsock
#include <stdio.h>
#include <string.h>
#include <winsock2.h>
#pragma comment(lib, "ws2_32.lib")

#define SERVER_IP   "127.0.0.1"
#define SERVER_PORT 7000
#define BUFSIZE     1024

int main(void) {
    WSADATA wsaData;
    SOCKET sock;
    struct sockaddr_in serverAddr;
    char buffer[BUFSIZE];
    int recvBytes;

    if (WSAStartup(MAKEWORD(2, 2), &wsaData) != 0) return 1;

    sock = socket(AF_INET, SOCK_STREAM, 0);
    if (sock == INVALID_SOCKET) { WSACleanup(); return 1; }

    serverAddr.sin_family = AF_INET;
    serverAddr.sin_port   = htons(SERVER_PORT);
    serverAddr.sin_addr.s_addr = inet_addr(SERVER_IP);

    // 主动连接到服务器
    if (connect(sock, (struct sockaddr*)&serverAddr, sizeof(serverAddr)) == SOCKET_ERROR) {
        perror("connect"); closesocket(sock); WSACleanup(); return 1;
    }
    printf("Connected to %s:%d\n", SERVER_IP, SERVER_PORT);

    // 发送一条消息
    const char *msg = "Hello, server!";
    send(sock, msg, strlen(msg), 0);

    // 等待回复
    recvBytes = recv(sock, buffer, BUFSIZE - 1, 0);
    if (recvBytes > 0) {
        buffer[recvBytes] = '\0';
        printf("Server replied: %s\n", buffer);
    }

    closesocket(sock);
    WSACleanup();
    return 0;
}

5. UDP Server / Client 代码

UDP Server

// udp_server.c
#include <stdio.h>
#include <winsock2.h>
#pragma comment(lib, "ws2_32.lib")

int main(void) {
    WSADATA wsa;
    WSAStartup(MAKEWORD(2,2), &wsa);

    SOCKET s = socket(AF_INET, SOCK_DGRAM, 0);   // ← UDP 用 SOCK_DGRAM
    struct sockaddr_in addr;
    addr.sin_family = AF_INET;
    addr.sin_port   = htons(8000);
    addr.sin_addr.s_addr = htonl(INADDR_ANY);
    bind(s, (struct sockaddr*)&addr, sizeof(addr));

    char buf[1024];
    struct sockaddr_in clientAddr;
    int clientLen = sizeof(clientAddr);

    // UDP 没有 listen/accept，直接 recvfrom
    int n = recvfrom(s, buf, sizeof(buf)-1, 0,
                     (struct sockaddr*)&clientAddr, &clientLen);
    buf[n] = '\0';
    printf("UDP recv: %s\n", buf);

    sendto(s, "pong", 4, 0,
           (struct sockaddr*)&clientAddr, clientLen);

    closesocket(s);
    WSACleanup();
    return 0;
}

UDP Client

// udp_client.c
#include <stdio.h>
#include <winsock2.h>
#pragma comment(lib, "ws2_32.lib")

int main(void) {
    WSADATA wsa;
    WSAStartup(MAKEWORD(2,2), &wsa);

    SOCKET s = socket(AF_INET, SOCK_DGRAM, 0);

    struct sockaddr_in server;
    server.sin_family = AF_INET;
    server.sin_port   = htons(8000);
    server.sin_addr.s_addr = inet_addr("127.0.0.1");

    // UDP 不需要 connect，直接 sendto
    sendto(s, "ping", 4, 0,
           (struct sockaddr*)&server, sizeof(server));

    char buf[1024];
    int srvLen = sizeof(server);
    int n = recvfrom(s, buf, sizeof(buf)-1, 0,
                     (struct sockaddr*)&server, &srvLen);
    buf[n] = '\0';
    printf("UDP reply: %s\n", buf);

    closesocket(s);
    WSACleanup();
    return 0;
}

UDP 关键差异（必考）：

  • socket 类型用 SOCK_DGRAM 而不是 SOCK_STREAM
  • 不调用 listen()、accept()、connect()
  • 用 sendto() / recvfrom()（携带地址参数）而不是 send() / recv()

6. 完整 Echo Server + Client（多次收发循环）

典型考题："写一个 TCP echo server 和 client，server 把收到的消息原样回送，client 可以多次发送，输入 'quit' 退出。"

Echo Server

// echo_server.c — 循环回送，处理一个客户端
#include <stdio.h>
#include <string.h>
#include <winsock2.h>
#pragma comment(lib, "ws2_32.lib")

#define PORT 9000

int main(void) {
    WSADATA wsa; WSAStartup(MAKEWORD(2,2), &wsa);

    SOCKET srv = socket(AF_INET, SOCK_STREAM, 0);
    struct sockaddr_in addr = {0};
    addr.sin_family = AF_INET;
    addr.sin_addr.s_addr = htonl(INADDR_ANY);
    addr.sin_port = htons(PORT);
    bind(srv, (struct sockaddr*)&addr, sizeof(addr));
    listen(srv, 5);
    printf("Echo server on port %d\n", PORT);

    struct sockaddr_in cli; int cliLen = sizeof(cli);
    SOCKET c = accept(srv, (struct sockaddr*)&cli, &cliLen);

    char buf[1024]; int n;
    while ((n = recv(c, buf, sizeof(buf)-1, 0)) > 0) {
        buf[n] = '\0';
        printf("Got: %s", buf);
        send(c, buf, n, 0);              // 原样回送
        if (strncmp(buf, "quit", 4) == 0) break;
    }

    closesocket(c); closesocket(srv); WSACleanup();
    return 0;
}

Echo Client

// echo_client.c — 反复读用户输入，发送给 server
#include <stdio.h>
#include <string.h>
#include <winsock2.h>
#pragma comment(lib, "ws2_32.lib")

int main(void) {
    WSADATA wsa; WSAStartup(MAKEWORD(2,2), &wsa);

    SOCKET s = socket(AF_INET, SOCK_STREAM, 0);
    struct sockaddr_in srv;
    srv.sin_family = AF_INET;
    srv.sin_port = htons(9000);
    srv.sin_addr.s_addr = inet_addr("127.0.0.1");
    connect(s, (struct sockaddr*)&srv, sizeof(srv));

    char line[1024], reply[1024];
    while (1) {
        printf("> "); fflush(stdout);
        if (!fgets(line, sizeof(line), stdin)) break;
        send(s, line, strlen(line), 0);
        int n = recv(s, reply, sizeof(reply)-1, 0);
        if (n <= 0) break;
        reply[n] = '\0';
        printf("Echo: %s", reply);
        if (strncmp(line, "quit", 4) == 0) break;
    }

    closesocket(s); WSACleanup();
    return 0;
}

7. 常考小题速记

字节序 Byte Order

网络字节序 = 大端 Big-Endian（高字节在低地址）。
x86 主机字节序 = 小端 Little-Endian。所以发送前要转换：

函数含义
htons(x)Host TO Network Short — 16 位（端口号）
htonl(x)Host TO Network Long — 32 位（IP 地址）
ntohs(x)Network TO Host Short
ntohl(x)Network TO Host Long

地址结构 sockaddr_in

struct sockaddr_in {
    short          sin_family;   // AF_INET
    unsigned short sin_port;     // 端口号（网络字节序）
    struct in_addr sin_addr;     // IP 地址（网络字节序）
    char           sin_zero[8];  // 填充，固定为 0
};

// 用法：
addr.sin_family = AF_INET;
addr.sin_port   = htons(8080);
addr.sin_addr.s_addr = inet_addr("192.168.1.1");
// 或绑定到所有网卡：
addr.sin_addr.s_addr = htonl(INADDR_ANY);

多线程并发服务器 Multi-threaded Server

#include <process.h>

void handle_client(void *arg) {
    SOCKET c = (SOCKET)(intptr_t)arg;
    // 处理这个客户端...
    closesocket(c);
    _endthread();
}

// main loop:
while (1) {
    SOCKET c = accept(srv, NULL, NULL);
    _beginthread(handle_client, 0, (void*)(intptr_t)c);
}

setsockopt — 允许端口快速重用

int opt = 1;
setsockopt(srv, SOL_SOCKET, SO_REUSEADDR, (char*)&opt, sizeof(opt));
// 防止 TIME_WAIT 期间端口"被占用"的报错

🎯 Self-Test 章末自测 (6 题)

Q1. TCP Server 的正确调用顺序是？

  • A. socket → connect → send → recv → close
  • B. socket → bind → accept → listen → recv → close
  • C. socket → listen → bind → accept → recv → close
  • D. socket → bind → listen → accept → recv/send → close
D. 必须先 bind 绑定端口，再 listen 转被动监听态，再 accept 等连接。

Q2. 创建 UDP socket 应使用：

  • A. socket(AF_INET, SOCK_DGRAM, 0)
  • B. socket(AF_INET, SOCK_STREAM, 0)
  • C. socket(AF_INET, SOCK_RAW, 0)
  • D. socket(AF_UDP, 0, 0)
A. UDP 用 SOCK_DGRAM；TCP 用 SOCK_STREAM。

Q3. htons(8080) 的作用是？

  • A. 把端口 8080 转 ASCII 字符串
  • B. 计算校验和
  • C. 把 16 位端口从主机字节序转换为网络字节序
  • D. 把字符串转 16 进制
C. Host TO Network Short：x86 是小端，网络是大端，必须转换。

Q4. 下列哪个调用在 UDP 中不会用到？

  • A. socket()
  • B. listen()
  • C. bind()
  • D. sendto()
B. UDP 无连接，不需要 listen/accept。仅 socket + bind + sendto/recvfrom + close。

Q5. 在 server 端 addr.sin_addr.s_addr = htonl(INADDR_ANY) 的含义？

  • A. 只允许本机连接
  • B. 限定 127.0.0.1
  • C. 绑定到本机所有网卡的 IP
  • D. 随机选择一个 IP
C. INADDR_ANY (0.0.0.0) 表示"任意网卡上的端口都接收"，多网卡机器常用。

Q6. 简答：在 TCP server 中，accept() 返回的新 socket 与原 server socket 有何区别？

参考答案：
  • 原 server socket：保持监听状态，继续接受新的连接请求
  • accept 返回的新 socket：专门和当前这个客户端通信，绑定的是已建立的连接（含双方 IP+Port 五元组）
  • 这种设计让 server 一边能继续 accept 新客户端，一边能通过新 socket 与已连接的客户端收发数据（通常 accept 后 fork/创建线程处理）
